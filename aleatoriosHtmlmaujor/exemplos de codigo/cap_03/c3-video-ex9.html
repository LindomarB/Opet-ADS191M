JavaScript

<script type="text/javascript">
function id(id) {
	return document.getElementById(id)
}
function init() {	 
	var video = id('video');
	var valor = id('valor');
	var btn = id('btn');
	var endereco = video.currentSrc;
	
btn.onclick = function(){
	var texto = 'Endereço do arquivo de mídia: ' + endereco;
	valor.innerHTML = (texto);
}
}
window.onload = init;
</script >
